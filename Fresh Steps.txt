CREATE TABLE cliente (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    nombre VARCHAR(100),
    apellido VARCHAR(100),
    correo VARCHAR(150),
    telefono VARCHAR(20),
    direccion VARCHAR(255)
);

CREATE TABLE servicio (
    id_servicio INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    descripcion TEXT,
    precio DECIMAL(10,2)
);

CREATE TABLE zapato (
    id_zapato INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    color_base VARCHAR(50),
    color_secundario VARCHAR(50),
    material VARCHAR(50),
    FOREIGN KEY (id_cliente) REFERENCES cliente(id_cliente)
);

CREATE TABLE venta (
    id_venta INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    fecha_recibo DATETIME DEFAULT CURRENT_TIMESTAMP,
    fecha_entrega DATETIME,
    tipo_pago VARCHAR(50),
    total DECIMAL(10,2),
    FOREIGN KEY (id_cliente) REFERENCES cliente(id_cliente)
);


CREATE TABLE venta_zapato (
    id_venta_zapato INT AUTO_INCREMENT PRIMARY KEY,
    id_venta INT,
    id_zapato INT,
    FOREIGN KEY (id_venta) REFERENCES venta(id_venta),
    FOREIGN KEY (id_zapato) REFERENCES zapato(id_zapato)
);


CREATE TABLE zapato_servicio (
    id_zapato_servicio INT AUTO_INCREMENT PRIMARY KEY,
    id_venta_zapato INT,
    id_servicio INT,
    precio_aplicado DECIMAL(10,2),
    FOREIGN KEY (id_venta_zapato) REFERENCES venta_zapato(id_venta_zapato),
    FOREIGN KEY (id_servicio) REFERENCES servicio(id_servicio)
);
